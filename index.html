<script>

function calc() {

  const role = document.getElementById('role').value;

  const n = Number(document.getElementById('numHires').value);


  const usSalary = Number(document.getElementById('usSalary').value);

  const latamSalary = Number(document.getElementById('latamSalary').value);


  const usB = (Number(document.getElementById('usBenefits').value) / 100) * usSalary;

  const laB = (Number(document.getElementById('latamBenefits').value) / 100) * latamSalary;


  const usTTH = Number(document.getElementById('usTTH').value);

  const latamTTH = Number(document.getElementById('latamTTH').value);


  const revenueImpact = Number(document.getElementById('revenueImpact').value);


  const annualSavingsPerRole = (usSalary + usB) - (latamSalary + laB);

  const annualSavings = annualSavingsPerRole * n;


  const dailyCoV = revenueImpact / 250;

  const timeSavedDays = Math.max(0, (usTTH - latamTTH) * 5);

  const covSaved = dailyCoV * timeSavedDays * n;


  const latamTotalCost = (latamSalary + laB) * n;

  const roi = latamTotalCost > 0 ? ((annualSavings + covSaved) / latamTotalCost) * 100 : 0;


  document.getElementById('annualSavings').innerText = "$" + annualSavings.toLocaleString();

  document.getElementById('covSaved').innerText = "$" + covSaved.toLocaleString();

  document.getElementById('roi').innerText = roi.toFixed(1) + "%";


  updateChart(annualSavings, covSaved, latamTotalCost);

}


let chart;


function updateChart(a, b, c) {

  const ctx = document.getElementById('chart').getContext('2d');

  if (chart) chart.destroy();


  chart = new Chart(ctx, {

    type: 'bar',

    data: {

      labels: ['Annual Savings', 'CoV Saved', 'LatAm Cost'],

      datasets: [{

        label: 'USD',

        data: [a, b, c],

        backgroundColor: ['#4f46e5', '#10b981', '#6b7280']

      }]

    },

    options: {

      responsive: true,

      scales: {

        y: { beginAtZero: true }

      }

    }

  });

}


function downloadCSV() {

  const rows = [

    ["Annual Savings", document.getElementById('annualSavings').innerText],

    ["CoV Saved", document.getElementById('covSaved').innerText],

    ["ROI", document.getElementById('roi').innerText]

  ];


  let csv = rows.map(r => r.join(",")).join("\n");

  const blob = new Blob([csv], { type: "text/csv" });

  const url = window.URL.createObjectURL(blob);

  const a = document.createElement("a");

  a.href = url;

  a.download = "roi_results.csv";

  a.click();

  window.URL.revokeObjectURL(url);

}


document.querySelectorAll("input").forEach(i => {

  i.addEventListener("input", calc);

});


calc();
